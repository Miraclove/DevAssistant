 13def get_files_from_directory(directory,file_type):
 14    """
 15    Returns a list of all files with the specified file type in the given directory and its subdirectories.
 16
 17    Args:
 18        directory (str): The directory to search for files.
 19        file_type (str): The file extension to search for (e.g. '.lua').
 20
 21    Returns:
 22        A list of file paths (str) for all files with the specified file type in the given directory and its subdirectories.
 23    """
 24    json_files = []
 25    for root, dirs, files in os.walk(directory):
 26        for file in files:
 27            if file.endswith(file_type):
 28                json_files.append(os.path.join(root, file))
 29    return json_files

 53def extract_info(h3_tag):
 54    """
 55    Extracts information from a given h3 tag and returns a dictionary containing the extracted information.
 56
 57    Args:
 58        h3_tag (bs4.element.Tag): The h3 tag to extract information from.
 59
 60    Returns:
 61        dict: A dictionary containing the extracted information. The dictionary may contain the following keys:
 62            - 'args': A list of dictionaries containing information about the function's arguments. Each dictionary may contain the following keys:
 63                - 'name': The name of the argument.
 64                - 'type': The type of the argument.
 65                - 'comment': A comment about the argument.
 66            - 'return': A dictionary containing information about the function's return value. The dictionary may contain the following keys:
 67                - 'type': The type of the return value.
 68                - 'comment': A comment about the return value.
 69            - 'example': A string containing an example of how to use the function.
 70    """
 71    result = {}
 72    
 73    next_element = h3_tag.find_next_sibling()
 74    if next_element and next_element.name == 'ul':
 75        # Extracting args
 76        args_list = []
 77        args_table = next_element.find_next_sibling('table')
 78        if args_table:
 79            for row in args_table.tbody.find_all('tr'):
 80                columns = [col.get_text(strip=True) for col in row.find_all('td')]
 81                if len(columns) == 3 and columns[0] != '-':
 82                    arg = {'name': columns[0], 'type': columns[1], 'comment': columns[2]}
 83                    args_list.append(arg)
 84            result['args'] = args_list
 85            next_element = args_table.find_next_sibling('ul')
 86
 87        # Extracting return value and example
 88        ret_tag = next_element.find('li')
 89        if ret_tag and 'ret:' in ret_tag.text:
 90            code_tag = ret_tag.find('code')
 91            if code_tag:
 92                result['return'] = {"type":code_tag.text.strip(),"comment":ret_tag.contents[2].strip()}
 93
 94        ret_tag = next_element.find('p')
 95        if ret_tag and 'ret:' in ret_tag.text:
 96            code_tag = ret_tag.find('code')
 97            if code_tag:
 98                result['return'] = {"type":code_tag.text.strip(),"comment":ret_tag.contents[2].strip()}
 99
100        
101
102    # Extracting  example
103    next_h3 = h3_tag.find_next_sibling('h3')
104    example_code = None
105    for sibling in h3_tag.find_all_next():
106        if sibling == next_h3:
107            break
108        if sibling.name == 'pre':
109            example_code = sibling
110            break
111
112    if example_code and example_code.code:
113        result['example'] = example_code.code.get_text(strip=True)
114    else:
115        result['example'] = None
116
117    return result

121def md_to_json(file_path):
122    """
123    Converts a markdown file to a JSON object containing information about events.
124
125    Args:
126        file_path (str): The path to the markdown file.
127
128    Returns:
129        dict: A dictionary containing information about events, with the following keys:
130            - 'events': A list of dictionaries, where each dictionary contains information about a single event.
131                Each event dictionary has the following keys:
132                    - 'description': A string describing the event.
133                    - 'function_name': The name of the function associated with the event.
134                    - Additional keys with information about the event, such as 'parameters' and 'returns'.
135    """
136    with open(file_path, 'r') as f:
137        md_file = f.read()
138
139    html_content = gfm(md_file)
140
141    soup = BeautifulSoup(html_content, 'html.parser')
142
143    events = []
144
145    for h3 in soup.find_all('h3'):
146        event_dict = {}
147        
148        # Getting the event's description and code name
149        event_dict['description'] = h3.contents[0].strip()
150        event_dict['function_name'] = h3.code.text.strip()
151        event_dict.update(extract_info(h3))
152        
153        events.append(event_dict)
154
155    json_output = {'events': events}
156    return json_output

288def find_comments_positions(lua_code):
289    """
290    Returns the positions of multi-line and single-line comments in the given Lua code.
291
292    Args:
293        lua_code (str): The Lua code to search for comments in.
294
295    Returns:
296        list: A list of tuples representing the start and end positions of each comment.
297              Each tuple contains two integers: the start position and the end position.
298              The list contains both single-line and multi-line comments.
299    """
300    single_line_comments = [(m.start(), m.end()) for m in re.finditer(r'--.*$', lua_code, re.MULTILINE)]
301    multi_line_comments = [(m.start(), m.end()) for m in re.finditer(r'--\[\[[\s\S]*?\]\]', lua_code)]
302    return single_line_comments + multi_line_comments

304def is_within_comment(position, comments):
305    """
306    Check if a given position is within a comment.
307
308    Args:
309        position (int): The position to check.
310        comments (list): A list of tuples representing the start and end positions of comments.
311
312    Returns:
313        bool: True if the position is within a comment, False otherwise.
314    """
315    for start, end in comments:
316        if start <= position <= end:
317            return True
318    return False

321def extract_function_names(lua_code):
322    """
323    Extracts the names of all functions defined in the given Lua code.
324
325    Args:
326        lua_code (str): The Lua code to extract function names from.
327
328    Returns:
329        list: A list of strings, where each string is the name of a function defined in the Lua code.
330    """
331    # This regex pattern matches both "function xxx(...)" and "local function xxx(...)"
332    pattern = re.compile(r'\b(?:local\s+)?function\s+([\w_:.]+)\s*?\(')
333    matches = pattern.findall(lua_code)
334    return matches

336def find_function_end(lua_code, start_index):
337    """
338    Finds the end of a Lua function starting from the given index in the code.
339
340    Args:
341        lua_code (str): The Lua code to search in.
342        start_index (int): The index to start searching from.
343
344    Returns:
345        int: The index of the end of the function, or -1 if not found.
346    """
347    stack = []
348    words = re.finditer(r'\b(function|if|for|while|end)\b', lua_code[start_index:])
349    comments = find_comments_positions(lua_code)
350    
351    for match in words:
352        if is_within_comment(match.start() + start_index, comments):
353            continue
354        
355        word = match.group(0)
356        if word in ["function","for","if","while"]:
357            stack.append(word)
358        elif word == "end":
359            stack.pop()
360        
361        if not stack:
362            return start_index + match.end()
363
364    return -1

366def extract_functions(lua_code):
367    """
368    Extracts all the functions from the given Lua code and returns them as a list of dictionaries.
369    
370    Each dictionary contains the following keys:
371    - 'name': the name of the function
372    - 'body': the body of the function
373    - 'comment': any comments preceding the function
374    
375    Args:
376    - lua_code (str): the Lua code to extract functions from
377    
378    Returns:
379    - A dictionary with a single key 'functions' whose value is a list of dictionaries, each representing a function.
380    """
381    results = []
382    comments = find_comments_positions(lua_code)
383    
384    for match in re.finditer(r'\b(?:local\s+)?function\b', lua_code):
385        # If the function keyword is within a comment, skip it
386        if is_within_comment(match.start(), comments):
387            continue
388        
389        # Go backwards from the function start to capture any preceding comments
390        
391        comment_start = match.start()
392        while comment_start > 0 and (lua_code[comment_start-1] == '\n' or is_within_comment(comment_start-1, comments)):
393            comment_start -= 1
394        
395        # extract function comments
396        if comment_start != match.start():
397            function_comment_part = lua_code[comment_start:match.start()]
398        else:
399            function_comment_part = ''
400            
401
402        # extract function body
403        start_index = match.start()
404        end_index = find_function_end(lua_code, match.start())
405        
406        if end_index != -1:
407            function_name = extract_function_names(lua_code[start_index:end_index])[0]
408            results.append({'name':function_name,'body':lua_code[start_index:end_index],'comment':function_comment_part})
409        else:
410            print("Error: Matching end not found for function starting at index", start_index)
411    return {'functions':results}

552def find_substring_in_file(filename, substring):
553    """Search for a substring in a file and return the line numbers where it's found."""
554    locations = []
555    with open(filename, 'r') as file:
556        for line_num, line in enumerate(file, 1):
557            if substring in line:
558                locations.append(line_num)
559    return (filename, locations) if locations else None

561def search_files_for_substring(directory, substring):
562    """Search for a substring in all Lua files in a directory and its subdirectories using multithreading."""
563    lua_files = [os.path.join(dirpath, filename)
564                 for dirpath, dirnames, filenames in os.walk(directory)
565                 for filename in filenames if filename.endswith('.lua')]
566    
567    # Limit the number of threads to 8
568    with concurrent.futures.ThreadPoolExecutor(max_workers=8) as executor:
569        results = list(executor.map(find_substring_in_file, lua_files, [substring]*len(lua_files)))
570
571    return [result for result in results if result]

593def extract_functions(lua_code,mute=False):
594    results = []
595    comments = find_comments_positions(lua_code)
596    
597    for match in re.finditer(r'\b(?:local\s+)?function\b', lua_code):
598        # If the function keyword is within a comment, skip it
599        if is_within_comment(match.start(), comments):
600            continue
601        
602        # Go backwards from the function start to capture any preceding comments
603        comment_start = match.start()
604        while comment_start > 0 and (lua_code[comment_start-1] == '\n' or is_within_comment(comment_start-1, comments)):
605            comment_start -= 1
606        
607        # extract function comments
608        if comment_start != match.start():
609            function_comment_part = lua_code[comment_start:match.start()]
610        else:
611            function_comment_part = ''
612            
613
614        # extract function body
615        start_index = match.start()
616        end_index = find_function_end(lua_code, match.start())
617        
618        if end_index != -1:
619            result = extract_function_names(lua_code[start_index:end_index])
620            if result:
621                function_name = result[0]
622            else:
623                continue
624            function_body = lua_code[start_index:end_index]
625            
626            # Check the lines after the function ends for the desired pattern
627            next_lines_start = end_index
628            while next_lines_start < len(lua_code) and lua_code[next_lines_start] not in ['\n', '\r']:
629                next_lines_start += 1
630            next_lines_end = next_lines_start + 1
631            while next_lines_end < len(lua_code) and lua_code[next_lines_end] not in ['\n', '\r']:
632                next_lines_end += 1
633            next_line = lua_code[next_lines_start:next_lines_end].strip()
634            register = None
635            if re.match(r'((GameEvent|g_mir.ssr)\.(add|register_ssr|remove|removeByNameAndTag|removeByTag|removeAll|push|register|execute|addEvent)\(.+\))', next_line):
636                register = next_line
